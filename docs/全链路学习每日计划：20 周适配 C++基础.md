### 核心文件示例：app/api/[recommend.py](recommend.py)

```Python

from fastapi import APIRouter, HTTPException
from app.models import RecommendRequest, RecommendResponse
from app.utils.model_loader import deepfm_model
from app.utils.feature_loader import get_user_features, get_item_features
import torch

router = APIRouter()

@router.post("/rank", response_model=RecommendResponse)
def rank_items(request: RecommendRequest):
    """
    精排接口：接收召回的物品ID，用DeepFM模型排序，返回Top-N
    """
    # 获取用户特征
    user_features = get_user_features(request.user_id)
    if not user_features:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    # 获取物品特征
    item_features_list = get_item_features(request.recall_item_ids)
    if not item_features_list:
        raise HTTPException(status_code=404, detail="物品不存在")
    
    # 模型推理
    model_input = torch.tensor([user_features + item_features for item_features in item_features_list], dtype=torch.float32)
    with torch.no_grad():
        ctr_scores = deepfm_model(model_input).squeeze().numpy()
    
    # 按CTR分数排序，取Top-N
    item_score_pairs = list(zip(request.recall_item_ids, ctr_scores))
    item_score_pairs.sort(key=lambda x: x[1], reverse=True)
    top_item_ids = [item_id for item_id, score in item_score_pairs[:request.top_n]]
    
    return {"user_id": request.user_id, "recommend_item_ids": top_item_ids}
```

package api

import (
 "backend-go/internal/milvus"
 "backend-go/internal/redis"
 "github.com/gin-gonic/gin"
 "net/http"
)

type RecallRequest struct {
 UserID int64 `json:"user_id" binding:"required"`
 TopN   int   `json:"top_n" default:"100"`
}

type RecallResponse struct {
 UserID        int64   `json:"user_id"`
 RecallItemIDs []int64 `json:"recall_item_ids"`
}

// VectorRecall 向量召回接口
func VectorRecall(c *gin.Context) {
 var req RecallRequest
 if err := c.ShouldBindJSON(&req); err != nil {
 c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
 return
 }

 // 从Redis获取用户最近点击的物品ID
 latestItemID, err := redis.GetUserLatestClick(req.UserID)
 if err != nil {
 c.JSON(http.StatusNotFound, gin.H{"error": "用户无历史行为"})
 return
 }

 // 从Milvus搜索相似物品
 recallItemIDs, err := milvus.SearchSimilarItems(latestItemID, req.TopN)
 if err != nil {
 c.JSON(http.StatusInternalServerError, gin.H{"error": "召回失败"})
 return
 }

 c.JSON(http.StatusOK, RecallResponse{
 UserID:        req.UserID,
 RecallItemIDs: recallItemIDs,
 })
}
